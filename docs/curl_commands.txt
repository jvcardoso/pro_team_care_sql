  -H "Content-Type: application/json" \
  -H "Authorization: Bearer SEU_TOKEN_AQUI" \
  -d '{
    "name": "Empresa Teste Manual",
    "tax_id": "99988877000111",
    "website": "https://manual.com",
    "description": "Teste manual via curl"
  }' \
  -v
